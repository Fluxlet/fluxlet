Read 'fluxlet.' prefixed params from the given strings 'fluxlet.=' will apply to any fluxlet 'fluxlet..=' will apply only to the specific fluxlet by id the fluxlet specific value will override the general value

function readUrlParams(id, strs) { const params = {}; function readParams(prefix) { strs.forEach(str => { const re = new RegExp(`\\b${prefix}\\.([^&;=#?.]+)=([^&;=#?]+)`, 'g'); let m; while (m = re.exec(str)) { params[m[1]] = m[2]; } }); } readParams('fluxlet'); id && readParams('fluxlet\\.' + id.replace(/[\\^$*+?.()|[\]{}]/g, '\\$&')); return params; }
